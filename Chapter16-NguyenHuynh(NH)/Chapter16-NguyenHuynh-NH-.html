"A leap year is a year that is either:
   1. Divisible for 4 and not divisible for 100
 or:
   2. Divisible for both 400 and 100
"
isLeapYear <- function(year) {
  if((year %% 4) == 0) {
    if((year %% 100) == 0) {
      if((year %% 400) == 0) {
        print("This is a leap year")
      }
      else {
        print("This is not a leap year")
      }
    }
    else {
      print("This is a leap year")
    }
  }
  else {
    print("This is not a leap year")
  }
}
